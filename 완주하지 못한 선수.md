### notice
array.slice(start,end) 는 인덱스 변화 X, 값 삭제<Br/>
array.splice(start,end)는 인덱스 변화 O, 값 삭제

#### 1 효율성 zero 
```javascript
function solution(participant, completion) {
    
    let i,j;
    let p = participant;
    let c = completion;
    
    while(c.length > 0){
        i=0;
        j=0;
        let check = c[i];
        
        for(j; j<p.length; j++){
            if(c[i]==p[j]){
                c.splice(i,1);
                p.splice(j,1);
                break;
            }
        }
    }
    
    var answer = '';
    answer = p.join('');
    
    return answer;
}
```
